31TArray<FIVSmokeHoleGPU> FIVSmokeHoleArray::GetHoleGPUData(const float CurrentServerTime) const
32{
33 TArray<FIVSmokeHoleGPU> GPUBuffer;
34 TArray<FIVSmokeHoleGPU> BulletBuffer;
35 TArray<FIVSmokeHoleGPU> GrenadeBuffer;
36 TArray<FIVSmokeHoleGPU> DynamicObjectBuffer;
37
38 BulletBuffer.Reserve(FMath::Max(Num(), 1));
39 GrenadeBuffer.Reserve(FMath::Max(Num(), 1));
40 DynamicObjectBuffer.Reserve(FMath::Max(Num(), 1));
41 GPUBuffer.Reserve(FMath::Max(Num(), 1));
42
43 for (const FIVSmokeHoleData& Hole : Items)
44 {
45 TObjectPtr<UIVSmokeHolePreset> Preset = UIVSmokeHolePreset::FindByID(Hole.PresetID);
46 if (!Preset)
47 {
48 continue;
49 }
50
51 FIVSmokeHoleGPU GPUHole = FIVSmokeHoleGPU(Hole, *Preset.Get(), CurrentServerTime);
52
53 if (Preset->HoleType == EIVSmokeHoleType::Penetration)
54 {
55 BulletBuffer.Add(GPUHole);
56 }
57 else if (Preset->HoleType == EIVSmokeHoleType::Explosion)
58 {
59 GrenadeBuffer.Add(GPUHole);
60 }
61 else if (Preset->HoleType == EIVSmokeHoleType::Dynamic)
62 {
63 DynamicObjectBuffer.Add(GPUHole);
64 }
65 }
66
67 GPUBuffer.Append(GrenadeBuffer);
68 GPUBuffer.Append(BulletBuffer);
69 GPUBuffer.Append(DynamicObjectBuffer);
70
71 if (GPUBuffer.Num() == 0)
72 {
73 GPUBuffer.AddDefaulted(1);
74 }
75
76 return GPUBuffer;
77}

79FIVSmokeHoleGPU::FIVSmokeHoleGPU(const FIVSmokeHoleData& DynamicHoleData, const UIVSmokeHolePreset& Preset, const float CurrentServerTime)
80{
81 Position = FVector3f(DynamicHoleData.Position);
82 EndPosition = FVector3f(DynamicHoleData.EndPosition);
83
84 HoleType = static_cast<int32>(Preset.HoleType);
85 Radius = Preset.Radius;
86 Duration = Preset.Duration;
87 if (Duration == 0)
88 {
89 return;
90 }
91 Softness = Preset.Softness;
92 ExpansionDuration = Preset.ExpansionDuration;
93
94 //SetTime
95 float RemainingTime = DynamicHoleData.ExpirationServerTime - CurrentServerTime;
96 CurLifeTime = Duration - RemainingTime;
97 float NormalizedTime = FMath::Clamp(CurLifeTime / Duration, 0.0f, 1.0f);
98
99 switch (Preset.HoleType)
100 {
101 case EIVSmokeHoleType::Explosion:
102 {
103 float ExpansionNormalizedTime = FMath::Clamp(CurLifeTime / Preset.ExpansionDuration, 0.0f, 1.0f);
104 float ShrinkNormalizedTime = FMath::Clamp((CurLifeTime - Preset.ExpansionDuration) / (Preset.Duration - Preset.ExpansionDuration), 0.0f, 1.0f);
105
106 CurExpansionFadeRangeOverTime = UIVSmokeHolePreset::GetFloatValue(Preset.ExpansionFadeRangeCurveOverTime, ExpansionNormalizedTime);
107 CurShrinkFadeRangeOverTime = UIVSmokeHolePreset::GetFloatValue(Preset.ShrinkFadeRangeCurveOverTime, ShrinkNormalizedTime);
108 CurShrinkDensityMulOverTime = UIVSmokeHolePreset::GetFloatValue(Preset.ShrinkDensityMulCurveOverTime, ShrinkNormalizedTime);
109 CurDistortionOverTime = UIVSmokeHolePreset::GetFloatValue(Preset.DistortionCurveOverTime, ExpansionNormalizedTime);
110 DistortionDistance = Preset.DistortionDistance;
111 UIVSmokeHolePreset::GetCurveSamples(Preset.DistortionCurveOverDistance.Get(), FIVSmokeHoleCarveCS::CurveSampleCount, DistortionCurveOverDistance);
112 break;
113 }
114 case EIVSmokeHoleType::Penetration:
115 EndRadius = Preset.EndRadius;
116 break;
117 case EIVSmokeHoleType::Dynamic:
118 Extent = Preset.Extent;
119 break;
120 }
121}
