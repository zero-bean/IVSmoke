65FIntVector FIVSmokeRenderer::GetAtlasTexCount(const FIntVector& TexSize, const int32 TexCount, const int32 TexturePackInterval, const int32 TexturePackMaxSize)
66{
67 int QuotientX = TexturePackMaxSize / (TexSize.X + TexturePackInterval);
68 int QuotientY = TexturePackMaxSize / (TexSize.Y + TexturePackInterval);
69 int QuotientZ = TexturePackMaxSize / (TexSize.Z + TexturePackInterval);
70
71 FIntVector AtlasTexCount = FIntVector(1, 1, 1);
72 if (QuotientX < TexCount)
73 {
74 AtlasTexCount.X = QuotientX;
75 }
76 else
77 {
78 AtlasTexCount.X = TexCount;
79 }
80
81 int CurTexCount = TexCount / QuotientX + (TexCount % QuotientX == 0 ? 0 : 1);
82 if (QuotientY < CurTexCount)
83 {
84 AtlasTexCount.Y = QuotientY;
85 }
86 else
87 {
88 AtlasTexCount.Y = CurTexCount;
89 }
90
91 CurTexCount = CurTexCount / QuotientY + (CurTexCount % QuotientY == 0 ? 0 : 1);
92 if (QuotientZ < CurTexCount)
93 {
94 // Warning: atlas size full
95 AtlasTexCount.Z = QuotientZ;
96 }
97 else
98 {
99 AtlasTexCount.Z = CurTexCount;
100 }
101 return AtlasTexCount;
102}

156bool FIVSmokeRenderer::GetMainDirectionalLight(UWorld* World, FVector& OutDirection, FLinearColor& OutColor, float& OutIntensity)
157{
158 if (!World)
159 {
160 return false;
161 }
162
163 UDirectionalLightComponent* BestLight = nullptr;
164 int32 BestIndex = INT_MAX;
165
166 // Find the atmosphere sun light with lowest index (0 = sun, 1 = moon)
167 for (TActorIterator<ADirectionalLight> It(World); It; ++It)
168 {
169 UDirectionalLightComponent* LightComp = Cast<UDirectionalLightComponent>(It->GetLightComponent());
170 if (LightComp && LightComp->IsUsedAsAtmosphereSunLight())
171 {
172 int32 Index = LightComp->GetAtmosphereSunLightIndex();
173 if (Index < BestIndex)
174 {
175 BestIndex = Index;
176 BestLight = LightComp;
177 }
178 }
179 }
180
181 // Fallback: first DirectionalLight found
182 if (!BestLight)
183 {
184 for (TActorIterator<ADirectionalLight> It(World); It; ++It)
185 {
186 BestLight = Cast<UDirectionalLightComponent>(It->GetLightComponent());
187 if (BestLight)
188 {
189 break;
190 }
191 }
192 }
193
194 if (BestLight)
195 {
196 // Negate: Shader expects direction TOWARD the light, not FROM the light
197 OutDirection = -BestLight->GetComponentRotation().Vector();
198 OutColor = BestLight->GetLightColor();
199 OutIntensity = BestLight->Intensity;
200 return true;
201 }
202
203 return false;
204}

206void FIVSmokeRenderer::CreateNoiseVolume()
207{
208 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
209 const FIVSmokeNoiseSettings& NoiseSettings = Settings->NoiseSettings;
210
211 // Create volume texture
212 NoiseVolume = NewObject<UTextureRenderTargetVolume>();
213 NoiseVolume->AddToRoot(); // Prevent GC
214 NoiseVolume->Init(NoiseSettings.TexSize, NoiseSettings.TexSize, NoiseSettings.TexSize, EPixelFormat::PF_R16F);
215 NoiseVolume->bCanCreateUAV = true;
216 NoiseVolume->ClearColor = FLinearColor::Black;
217 NoiseVolume->SRGB = false;
218 NoiseVolume->UpdateResourceImmediate(true);
219
220 // Run compute shader to generate noise
221 FTextureRenderTargetResource* RenderTargetResource = NoiseVolume->GameThread_GetRenderTargetResource();
222 if (!RenderTargetResource)
223 {
224 UE_LOG(LogIVSmoke, Error, TEXT("[FIVSmokeRenderer::CreateNoiseVolume] Failed to get render target resource"));
225 return;
226 }
227
228 ENQUEUE_RENDER_COMMAND(IVSmokeGenerateNoise)(
229 [RenderTargetResource, NoiseSettings](FRHICommandListImmediate& RHICmdList)
230 {
231 FRDGBuilder GraphBuilder(RHICmdList);
232 FRDGTextureRef NoiseTexture = GraphBuilder.RegisterExternalTexture(
233 CreateRenderTarget(RenderTargetResource->TextureRHI, TEXT("IVSmokeNoiseVolume"))
234 );
235
236 FRDGTextureUAVRef OutputUAV = GraphBuilder.CreateUAV(NoiseTexture);
237
238 auto* Parameters = GraphBuilder.AllocParameters<FIVSmokeNoiseGeneratorGlobalCS::FParameters>();
239 Parameters->RWNoiseTex = OutputUAV;
240 Parameters->TexSize = FUintVector3(NoiseSettings.TexSize, NoiseSettings.TexSize, NoiseSettings.TexSize);
241 Parameters->Octaves = NoiseSettings.Octaves;
242 Parameters->Wrap = NoiseSettings.Wrap;
243 Parameters->AxisCellCount = NoiseSettings.AxisCellCount;
244 Parameters->Amplitude = NoiseSettings.Amplitude;
245 Parameters->CellSize = NoiseSettings.CellSize;
246 Parameters->Seed = NoiseSettings.Seed;
247
248 TShaderMapRef<FIVSmokeNoiseGeneratorGlobalCS> ComputeShader(GetGlobalShaderMap(GMaxRHIFeatureLevel));
249
250 FIntVector GroupCount(
251 FMath::DivideAndRoundUp(NoiseSettings.TexSize, 8),
252 FMath::DivideAndRoundUp(NoiseSettings.TexSize, 8),
253 FMath::DivideAndRoundUp(NoiseSettings.TexSize, 8)
254 );
255
256 GraphBuilder.AddPass(
257 RDG_EVENT_NAME("IVSmokeNoiseGeneration"),
258 Parameters,
259 ERDGPassFlags::Compute,
260 [Parameters, ComputeShader, GroupCount](FRHIComputeCommandList& RHICmdList)
261 {
262 FComputeShaderUtils::Dispatch(RHICmdList, ComputeShader, *Parameters, GroupCount);
263 }
264 );
265 GraphBuilder.Execute();
266 }
267 );
268}

316FIVSmokePackedRenderData FIVSmokeRenderer::PrepareRenderData(const TArray<AIVSmokeVoxelVolume*>& InVolumes)
317{
318 // Must be called on Game Thread
319 check(IsInGameThread());
320
321 FIVSmokePackedRenderData Result;
322
323 if (InVolumes.Num() == 0)
324 {
325 return Result;
326 }
327
328 Result.VolumeCount = InVolumes.Num();
329 Result.VolumeDataArray.Reserve(Result.VolumeCount);
330 Result.HoleTextures.Reserve(Result.VolumeCount);
331 Result.HoleTextureSizes.Reserve(Result.VolumeCount);
332
333 // Get resolution info from first valid volume
334 for (AIVSmokeVoxelVolume* Volume : InVolumes)
335 {
336 if (Volume)
337 {
338 Result.VoxelResolution = Volume->GetGridResolution();
339 if (UIVSmokeHoleGeneratorComponent* HoleComp = Volume->GetHoleGeneratorComponent())
340 {
341 if (FTextureRHIRef HoleTex = HoleComp->GetHoleTextureRHI())
342 {
343 Result.HoleResolution = HoleTex->GetSizeXYZ();
344 }
345 }
346 break;
347 }
348 }
349
350 // Fallback for hole resolution
351 if (Result.HoleResolution == FIntVector::ZeroValue)
352 {
353 Result.HoleResolution = FIntVector(64, 64, 64);
354 }
355
356 // Calculate packed buffer sizes
357 const int32 TexturePackInterval = 4;
358 TArray<float> VoxelIntervalData;
359 VoxelIntervalData.Init(0, Result.VoxelResolution.X * Result.VoxelResolution.Y * TexturePackInterval);
360
361 FIntVector VoxelAtlasResolution = FIntVector(
362 Result.VoxelResolution.X,
363 Result.VoxelResolution.Y,
364 Result.VoxelResolution.Z * Result.VolumeCount + TexturePackInterval * (Result.VolumeCount - 1)
365 );
366 int32 TotalVoxelSize = VoxelAtlasResolution.X * VoxelAtlasResolution.Y * VoxelAtlasResolution.Z;
367 Result.PackedVoxelBirthTimes.Reserve(TotalVoxelSize);
368 Result.PackedVoxelDeathTimes.Reserve(TotalVoxelSize);
369
370 // Collect data from all volumes (Game Thread - safe to access)
371 for (int32 i = 0; i < InVolumes.Num(); ++i)
372 {
373 AIVSmokeVoxelVolume* Volume = InVolumes[i];
374 if (!Volume)
375 {
376 continue;
377 }
378
379 //~==========================================================================
380 // Copy VoxelArray data (Game Thread safe)
381 const TArray<float>& VoxelBirthTimes = Volume->GetVoxelBirthTimes();
382 Result.PackedVoxelBirthTimes.Append(VoxelBirthTimes);
383
384 const TArray<float>& VoxelDeathTimes = Volume->GetVoxelDeathTimes();
385 Result.PackedVoxelDeathTimes.Append(VoxelDeathTimes);
386
387 if (i < InVolumes.Num() - 1)
388 {
389 Result.PackedVoxelBirthTimes.Append(VoxelIntervalData);
390 Result.PackedVoxelDeathTimes.Append(VoxelIntervalData);
391 }
392
393 //~==========================================================================
394 // Hole Texture reference (RHI resources are thread-safe)
395 if (UIVSmokeHoleGeneratorComponent* HoleComp = Volume->GetHoleGeneratorComponent())
396 {
397 FTextureRHIRef HoleTex = HoleComp->GetHoleTextureRHI();
398 Result.HoleTextures.Add(HoleTex);
399 if (HoleTex)
400 {
401 Result.HoleTextureSizes.Add(HoleTex->GetSizeXYZ());
402 }
403 else
404 {
405 Result.HoleTextureSizes.Add(FIntVector::ZeroValue);
406 }
407 }
408 else
409 {
410 Result.HoleTextures.Add(nullptr);
411 Result.HoleTextureSizes.Add(FIntVector::ZeroValue);
412 }
413
414 //~==========================================================================
415 // Build GPU metadata
416 const FIntVector GridRes = Volume->GetGridResolution();
417 const FIntVector CenterOff = Volume->GetCenterOffset();
418 const float VoxelSz = Volume->GetVoxelSize();
419 const FTransform VolumeTransform = Volume->GetActorTransform();
420
421 // Calculate AABB
422 FVector HalfExtent = FVector(CenterOff) * VoxelSz;
423 FVector LocalMin = -HalfExtent;
424 FVector LocalMax = FVector(GridRes - CenterOff - FIntVector(1, 1, 1)) * VoxelSz;
425 FBox LocalBox(LocalMin, LocalMax);
426 FBox WorldBox = LocalBox.TransformBy(VolumeTransform);
427
428 // Get preset data
429 const UIVSmokeSmokePreset* Preset = GetEffectivePreset(Volume);
430
431 // Build GPU data struct
432 FIVSmokeVolumeGPUData GPUData;
433 FMemory::Memzero(&GPUData, sizeof(GPUData));
434
435 GPUData.VoxelSize = VoxelSz;
436 GPUData.VoxelBufferOffset = Result.VoxelResolution.X * Result.VoxelResolution.Y *
437 (Result.VoxelResolution.Z + TexturePackInterval) * i;
438 GPUData.GridResolution = FIntVector3(GridRes.X, GridRes.Y, GridRes.Z);
439 GPUData.VoxelCount = VoxelBirthTimes.Num();
440 GPUData.CenterOffset = FVector3f(CenterOff.X, CenterOff.Y, CenterOff.Z);
441 GPUData.VolumeWorldAABBMin = FVector3f(WorldBox.Min);
442 GPUData.VolumeWorldAABBMax = FVector3f(WorldBox.Max);
443 GPUData.VoxelWorldAABBMin = FVector3f(Volume->GetVoxelWorldAABBMin());
444 GPUData.VoxelWorldAABBMax = FVector3f(Volume->GetVoxelWorldAABBMax());
445 GPUData.FadeInDuration = Volume->FadeInDuration;
446 GPUData.FadeOutDuration = Volume->FadeOutDuration;
447
448 if (Preset)
449 {
450 GPUData.SmokeColor = FVector3f(Preset->SmokeColor.R, Preset->SmokeColor.G, Preset->SmokeColor.B);
451 GPUData.Absorption = Preset->SmokeAbsorption;
452 GPUData.DensityScale = Preset->VolumeDensity;
453 }
454 else
455 {
456 GPUData.SmokeColor = FVector3f(0.8f, 0.8f, 0.8f);
457 GPUData.Absorption = 0.1f;
458 GPUData.DensityScale = 1.0f;
459 }
460
461 Result.VolumeDataArray.Add(GPUData);
462 }
463
464 //~==========================================================================
465 // Copy global settings parameters
466 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
467
468 if (Settings)
469 {
470 // Post processing
471 Result.Sharpness = Settings->Sharpness;
472
473 // Ray marching
474 Result.MaxSteps = Settings->GetEffectiveMaxSteps();
475
476 // Appearance
477 Result.GlobalAbsorption = 0.1f; // Default, per-volume absorption from preset
478 Result.SmokeSize = Settings->SmokeSize;
479 Result.SmokeDensityFalloff = Settings->SmokeDensityFalloff;
480 Result.WindDirection = Settings->WindDirection;
481 Result.VolumeRangeOffset = Settings->VolumeRangeOffset;
482 Result.VolumeEdgeNoiseFadeOffset = Settings->VolumeEdgeNoiseFadeOffset;
483 Result.VolumeEdgeFadeSharpness = Settings->VolumeEdgeFadeSharpness;
484
485 // Scattering
486 Result.bEnableScattering = Settings->bEnableScattering;
487 Result.ScatterScale = Settings->ScatterScale;
488 Result.ScatteringAnisotropy = Settings->ScatteringAnisotropy;
489
490 // Get world from first volume (single lookup, reused for light detection and shadow capture)
491 UWorld* World = (InVolumes.Num() > 0 && InVolumes[0]) ? InVolumes[0]->GetWorld() : nullptr;
492
493 // Light Direction and Color
494 // Priority: Settings Override > World DirectionalLight > Default
495 if (Settings->bOverrideLightDirection)
496 {
497 Result.LightDirection = Settings->LightDirectionOverride.GetSafeNormal();
498 Result.LightIntensity = 1.0f; // Override assumes full intensity
499 }
500 else
501 {
502 FVector AutoLightDir;
503 FLinearColor AutoLightColor;
504 float AutoLightIntensity;
505
506 if (GetMainDirectionalLight(World, AutoLightDir, AutoLightColor, AutoLightIntensity))
507 {
508 Result.LightDirection = AutoLightDir;
509 Result.LightIntensity = AutoLightIntensity;
510
511 // Also use auto light color if not overridden
512 if (!Settings->bOverrideLightColor)
513 {
514 Result.LightColor = AutoLightColor;
515 }
516 }
517 else
518 {
519 // No directional light found - dark environment
520 Result.LightDirection = FVector(0.0f, 0.0f, -1.0f);
521 Result.LightIntensity = 0.0f;
522 Result.LightColor = FLinearColor::Black;
523 }
524 }
525
526 if (Settings->bOverrideLightColor)
527 {
528 Result.LightColor = Settings->LightColorOverride;
529 }
530
531 // Self-shadowing
532 Result.bEnableSelfShadowing = Settings->bEnableSelfShadowing;
533 Result.LightMarchingSteps = Settings->LightMarchingSteps;
534 Result.LightMarchingDistance = Settings->LightMarchingDistance;
535 Result.LightMarchingExpFactor = Settings->LightMarchingExpFactor;
536 Result.ShadowAmbient = Settings->ShadowAmbient;
537
538 // External shadowing (CSM - Cascaded Shadow Maps)
539 // Note: CSM is always used when external shadowing is enabled. NumCascades > 0 indicates active.
540 Result.ShadowDepthBias = Settings->ShadowDepthBias;
541 Result.ExternalShadowAmbient = Settings->ExternalShadowAmbient;
542
543 // VSM settings
544 Result.bEnableVSM = Settings->bEnableVSM;
545 Result.VSMMinVariance = Settings->VSMMinVariance;
546 Result.VSMLightBleedingReduction = Settings->VSMLightBleedingReduction;
547 Result.CascadeBlendRange = Settings->CascadeBlendRange;
548
549 // Skip shadow capture if we're already inside a shadow capture render pass (prevents infinite recursion)
550 if (Settings->bEnableExternalShadowing && InVolumes.Num() > 0 && !bIsCapturingShadow)
551 {
552 // Per-frame guard: Only update once per actual engine frame
553 // PrepareRenderData can be called multiple times per frame (multiple views)
554 const uint32 CurrentFrameNumber = GFrameNumber;
555 const bool bAlreadyUpdatedThisFrame = (LastCSMUpdateFrameNumber == CurrentFrameNumber);
556
557 if (!bAlreadyUpdatedThisFrame && World)
558 {
559 LastCSMUpdateFrameNumber = CurrentFrameNumber;
560
561 // Initialize CSM if needed
562 InitializeCSM(World);
563
564 if (CSMRenderer && CSMRenderer->IsInitialized())
565 {
566 // Set re-entry guard (safety measure)
567 bIsCapturingShadow = true;
568
569 // Get camera position from first volume's world (or use centroid of volumes)
570 FVector CameraPosition = FVector::ZeroVector;
571 FVector CameraForward = FVector(1.0f, 0.0f, 0.0f);
572
573 // Try to get player camera position
574 if (APlayerController* PC = World->GetFirstPlayerController())
575 {
576 if (APlayerCameraManager* CameraManager = PC->PlayerCameraManager)
577 {
578 CameraPosition = CameraManager->GetCameraLocation();
579 CameraForward = CameraManager->GetCameraRotation().Vector();
580 }
581 }
582
583 // Update CSM with current frame
584 CSMRenderer->Update(
585 CameraPosition,
586 CameraForward,
587 Result.LightDirection,
588 CurrentFrameNumber
589 );
590
591 bIsCapturingShadow = false;
592 }
593 }
594
595 // Populate CSM data for shader (even if not updated this frame)
596 if (CSMRenderer && CSMRenderer->IsInitialized() && CSMRenderer->HasValidShadowData())
597 {
598 Result.NumCascades = CSMRenderer->GetNumCascades();
599
600 // Get split distances
601 Result.CSMSplitDistances = CSMRenderer->GetSplitDistances();
602
603 // Get textures, matrices, and light camera data for each cascade
604 Result.CSMDepthTextures.SetNum(Result.NumCascades);
605 Result.CSMVSMTextures.SetNum(Result.NumCascades);
606 Result.CSMViewProjectionMatrices.SetNum(Result.NumCascades);
607 Result.CSMLightCameraPositions.SetNum(Result.NumCascades);
608 Result.CSMLightCameraForwards.SetNum(Result.NumCascades);
609
610 for (int32 i = 0; i < Result.NumCascades; i++)
611 {
612 const FIVSmokeCascadeData& Cascade = CSMRenderer->GetCascade(i);
613 // Single-buffer: VP matrix and texture are from the SAME frame
614 Result.CSMViewProjectionMatrices[i] = Cascade.ViewProjectionMatrix;
615 Result.CSMDepthTextures[i] = CSMRenderer->GetDepthTexture(i);
616 Result.CSMVSMTextures[i] = CSMRenderer->GetVSMTexture(i);
617 Result.CSMLightCameraPositions[i] = Cascade.LightCameraPosition;
618 Result.CSMLightCameraForwards[i] = Cascade.LightCameraForward;
619 }
620
621 // Store the main camera position for consistent use in shader
622 Result.CSMMainCameraPosition = CSMRenderer->GetMainCameraPosition();
623 }
624 }
625 }
626
627 Result.bIsValid = Result.VolumeDataArray.Num() && Result.PackedVoxelBirthTimes.Num() > 0 && Result.PackedVoxelDeathTimes.Num() > 0;
628
629 if (InVolumes.Num() > 0 && InVolumes[0])
630 {
631 Result.GameTime = InVolumes[0]->GetSyncWorldTimeSeconds();
632 }
633 else
634 {
635 Result.GameTime = 0.0f;
636 }
637
638 return Result;
639}

644FScreenPassTexture FIVSmokeRenderer::Render(
645 FRDGBuilder& GraphBuilder,
646 const FSceneView& View,
647 const FPostProcessMaterialInputs& Inputs)
648{
649 // Get scene color from inputs FIRST - needed for passthrough
650 FScreenPassTextureSlice SceneColorSlice = Inputs.GetInput(EPostProcessMaterialInput::SceneColor);
651 if (!SceneColorSlice.IsValid())
652 {
653 return FScreenPassTexture();
654 }
655
656 FScreenPassTexture SceneColor(SceneColorSlice);
657
658 // Check if rendering is enabled - passthrough if disabled
659 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
660 if (!Settings->bEnableSmokeRendering)
661 {
662 return SceneColor;
663 }
664
665 FScreenPassRenderTarget Output = Inputs.OverrideOutput;
666
667 if (!Output.IsValid())
668 {
669 Output = FScreenPassRenderTarget(
670 SceneColor.Texture,
671 SceneColor.ViewRect,
672 ERenderTargetLoadAction::ELoad
673 );
674 }
675
676 // Use ViewRect size consistently for all passes
677 const FIntPoint ViewportSize = SceneColor.ViewRect.Size();
678 const FIntPoint ViewRectMin = SceneColor.ViewRect.Min;
679
680 //~==========================================================================
681 // Upscaling Pipeline (1/2 to Full)
682 //
683 // Ray March at 1/2 resolution for quality/performance balance.
684 // Single-step upscaling with bilinear filtering smooths IGN grain.
685 // Note: 1/4 resolution causes excessive grain when camera is inside smoke.
686 //
687 const FIntPoint HalfSize = FIntPoint(
688 FMath::Max(1, ViewportSize.X / 2),
689 FMath::Max(1, ViewportSize.Y / 2)
690 );
691
692 // Create Dual Render Target textures at 1/2 resolution
693 FRDGTextureRef SmokeAlbedoTex = FIVSmokePostProcessPass::CreateOutputTexture(
694 GraphBuilder,
695 SceneColor.Texture,
696 TEXT("IVSmokeAlbedoTex_Half"),
697 PF_FloatRGBA,
698 HalfSize
699 );
700
701 FRDGTextureRef SmokeMaskTex = FIVSmokePostProcessPass::CreateOutputTexture(
702 GraphBuilder,
703 SceneColor.Texture,
704 TEXT("IVSmokeMaskTex_Half"),
705 PF_FloatRGBA,
706 HalfSize
707 );
708
709 // Get cached render data (prepared on Game Thread via BeginRenderViewFamily)
710 // Use copy instead of MoveTemp - multiple views in same frame share the same data
711 FIVSmokePackedRenderData RenderData;
712 {
713 FScopeLock Lock(&RenderDataMutex);
714 RenderData = CachedRenderData; // Copy - don't consume, other views may need it
715 }
716
717 if (!RenderData.bIsValid)
718 {
719 return SceneColor;
720 }
721
722 //~==========================================================================
723 // Ray March Pass (1/2 Resolution)
724 // Multi-Volume Ray Marching with Occupancy Optimization (Three-Pass Pipeline).
725 // Uses tile-based occupancy grid for efficient empty space skipping.
726 AddMultiVolumeRayMarchPass(
727 GraphBuilder,
728 View,
729 RenderData,
730 SmokeAlbedoTex,
731 SmokeMaskTex,
732 HalfSize,
733 ViewportSize,
734 ViewRectMin
735 );
736
737 //~==========================================================================
738 // Upscaling (1/2 to Full)
739 // Single-step bilinear upscaling smooths IGN grain patterns.
740

742 FRDGTextureRef SmokeAlbedoFull = AddCopyPass(
743 GraphBuilder,

748 );

752 GraphBuilder,
753 View,
754 SmokeMaskTex,
755 ViewportSize,
756 TEXT("IVSmokeMaskTex_Full")
757 );
758
759 //~==========================================================================
760 // Composite Pass
761 const float Sharpness = RenderData.Sharpness;
762 const bool bUseCustomDepthBasedSorting = Settings->bUseCustomDepthBasedSorting;
763
764 // Check if we're in TranslucencyAfterDOF mode (setting + SeparateTranslucency input valid)
765 const bool bTranslucencyMode = (Settings->RenderPass == EIVSmokeRenderPass::TranslucencyAfterDOF);
766 FScreenPassTextureSlice SeparateTranslucencySlice = Inputs.GetInput(EPostProcessMaterialInput::SeparateTranslucency);
767
768 //~==========================================================================
769 // Depth-Sorted Composite: Proper smoke/particle sorting using CustomDepth
770 if (bUseCustomDepthBasedSorting && bTranslucencyMode && SeparateTranslucencySlice.IsValid())
771 {
772 FScreenPassTexture ParticlesTex(SeparateTranslucencySlice);
773
774 // Create output texture based on ParticlesTex (same as TranslucencyComposite)
775 // TranslucencyAfterDOF mode expects output in SeparateTranslucency format
776 FRDGTextureRef OutputTexture = FIVSmokePostProcessPass::CreateOutputTexture(
777 GraphBuilder,
778 ParticlesTex.Texture,
779 TEXT("IVSmokeDepthSortedOutput"),
780 PF_FloatRGBA
781 );
782
783 FScreenPassRenderTarget SortedOutput(
784 OutputTexture,
785 ParticlesTex.ViewRect,
786 ERenderTargetLoadAction::ENoAction
787 );
788
789 // Pass texture extents for UV calculation (UV = SvPosition / TexExtent)
790 const FIntPoint SmokeExtent = ViewportSize;
791
792 AddDepthSortedCompositePass(
793 GraphBuilder,
794 View,
795 SmokeAlbedoFull,
796 SmokeMaskFull,
797 ParticlesTex.Texture,
798 SortedOutput,
799 SmokeExtent,
800 Sharpness
801 );
802
803 return FScreenPassTexture(SortedOutput);
804 }
805 //~==========================================================================
806 // Standard TranslucencyAfterDOF Mode: Smoke OVER particles (no depth sorting)
807 else if (bTranslucencyMode && SeparateTranslucencySlice.IsValid())
808 {
809 // TranslucencyAfterDOF mode: Composite smoke OVER particles
810 FScreenPassTexture ParticlesTex(SeparateTranslucencySlice);
811
812 // Smoke textures are rendered at SceneColor.ViewRect
813 // Particles texture is at its own ViewRect (SeparateTranslucency)
814 // These can differ! Shader handles separate UV calculation for each.
815
816 // Create output texture with SAME SIZE as ParticlesTex
817 FRDGTextureRef OutputTexture = FIVSmokePostProcessPass::CreateOutputTexture(
818 GraphBuilder,
819 ParticlesTex.Texture,
820 TEXT("IVSmokeTranslucencyOutput"),
821 PF_FloatRGBA
822 );
823
824 FScreenPassRenderTarget TranslucencyOutput(
825 OutputTexture,
826 ParticlesTex.ViewRect,
827 ERenderTargetLoadAction::ENoAction
828 );
829
830 // Pass texture extents for UV calculation (UV = SvPosition / TexExtent)
831 const FIntPoint SmokeExtent = ViewportSize;
832 const FIntPoint ParticlesExtent(ParticlesTex.Texture->Desc.Extent.X, ParticlesTex.Texture->Desc.Extent.Y);
833
834 AddTranslucencyCompositePass(
835 GraphBuilder,
836 View,
837 SmokeAlbedoFull,
838 SmokeMaskFull,
839 ParticlesTex.Texture,
840 TranslucencyOutput,
841 SmokeExtent,
842 ParticlesExtent,
843 Sharpness
844 );
845
846 return FScreenPassTexture(TranslucencyOutput);
847 }
848 //~==========================================================================
849 // Standard Mode: Composite smoke with scene color
850 else
851 {
852 AddSharpenCompositePass(
853 GraphBuilder,
854 View,
855 SceneColor.Texture,
856 SmokeAlbedoFull,
857 SmokeMaskFull,
858 Output,
859 ViewportSize,
860 Sharpness
861 );
862
863 return FScreenPassTexture(Output);
864 }
865}

1016// Multi-Volume Ray March Pass (Occupancy-Based Three-Pass Pipeline)
1017
1018void FIVSmokeRenderer::AddMultiVolumeRayMarchPass(
1019 FRDGBuilder& GraphBuilder,
1020 const FSceneView& View,
1021 const FIVSmokePackedRenderData& RenderData,
1022 FRDGTextureRef SmokeAlbedoTex,
1023 FRDGTextureRef SmokeMaskTex,
1024 const FIntPoint& TexSize,
1025 const FIntPoint& ViewportSize,
1026 const FIntPoint& ViewRectMin)
1027{
1028 const int32 VolumeCount = RenderData.VolumeCount;
1029
1030 if (VolumeCount == 0 || !NoiseVolume || !RenderData.bIsValid)
1031 {
1032 return;
1033 }
1034
1035 // Get global settings
1036 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
1037
1038 //~==========================================================================
1039 // Phase 0: Setup common resources (same as standard ray march)
1040
1041 const int32 TexturePackInterval = 4;
1042 const int32 TexturePackMaxSize = 2048;
1043 const FIntVector VoxelResolution = RenderData.VoxelResolution;
1044 const FIntVector HoleResolution = RenderData.HoleResolution;
1045 const FIntVector HoleAtlasCount = GetAtlasTexCount(HoleResolution, VolumeCount, TexturePackInterval, TexturePackMaxSize);
1046
1047 // Voxel Atlas: 1D Z-stack (must match CollectVolumeRenderData packing)
1048 const FIntVector VoxelAtlasResolution = FIntVector(
1049 VoxelResolution.X,
1050 VoxelResolution.Y,
1051 VoxelResolution.Z * VolumeCount + TexturePackInterval * (VolumeCount - 1)
1052 );
1053 const FIntVector VoxelAtlasFXAAResolution = VoxelAtlasResolution * 1;
1054
1055 // Hole Atlas: 3D packing
1056 const FIntVector HoleAtlasResolution = FIntVector(
1057 HoleResolution.X * HoleAtlasCount.X + TexturePackInterval * (HoleAtlasCount.X - 1),
1058 HoleResolution.Y * HoleAtlasCount.Y + TexturePackInterval * (HoleAtlasCount.Y - 1),
1059 HoleResolution.Z * HoleAtlasCount.Z + TexturePackInterval * (HoleAtlasCount.Z - 1)
1060 );
1061
1062 // Create atlas textures
1063 FRDGTextureDesc VoxelAtlasDesc = FRDGTextureDesc::Create3D(
1064 VoxelAtlasResolution,
1065 PF_R32_FLOAT,
1066 FClearValueBinding::None,
1067 TexCreate_ShaderResource | TexCreate_UAV
1068 );
1069 FRDGTextureRef PackedVoxelAtlas = GraphBuilder.CreateTexture(VoxelAtlasDesc, TEXT("IVSmoke_PackedVoxelAtlas"));
1070
1071 FRDGTextureDesc VoxelAtlasFXAAResDesc = FRDGTextureDesc::Create3D(
1072 VoxelAtlasFXAAResolution,
1073 PF_R32_FLOAT,
1074 FClearValueBinding::None,
1075 TexCreate_ShaderResource | TexCreate_UAV
1076 );
1077 FRDGTextureRef PackedVoxelAtlasFXAA = GraphBuilder.CreateTexture(VoxelAtlasFXAAResDesc, TEXT("IVSmoke_PackedVoxelAtlasFXAA"));
1078
1079 FRDGTextureDesc HoleAtlasDesc = FRDGTextureDesc::Create3D(
1080 HoleAtlasResolution,
1081 PF_FloatRGBA,
1082 FClearValueBinding::None,
1083 TexCreate_ShaderResource | TexCreate_UAV
1084 );
1085 FRDGTextureRef PackedHoleAtlas = GraphBuilder.CreateTexture(HoleAtlasDesc, TEXT("IVSmoke_PackedHoleAtlas"));
1086
1087 // Clear Hole Atlas
1088 AddClearUAVPass(GraphBuilder, GraphBuilder.CreateUAV(PackedHoleAtlas), 0.0f);
1089
1090 // Copy Hole Textures to Atlas
1091 FRHICopyTextureInfo HoleCpyInfo;
1092 HoleCpyInfo.Size = HoleResolution;
1093 HoleCpyInfo.SourcePosition = FIntVector::ZeroValue;
1094
1095 for (int z = 0; z < HoleAtlasCount.Z; ++z)
1096 {
1097 for (int y = 0; y < HoleAtlasCount.Y; ++y)
1098 {
1099 for (int x = 0; x < HoleAtlasCount.X; ++x)
1100 {
1101 int i = x + HoleAtlasCount.X * y + z * HoleAtlasCount.X * HoleAtlasCount.Y;
1102
1103 if (i >= RenderData.HoleTextures.Num())
1104 {
1105 break;
1106 }
1107
1108 FTextureRHIRef SourceRHI = RenderData.HoleTextures[i];
1109 if (!SourceRHI)
1110 {
1111 continue;
1112 }
1113
1114 FRDGTextureRef SourceTexture = GraphBuilder.RegisterExternalTexture(
1115 CreateRenderTarget(SourceRHI, TEXT("IVSmoke_CopyHoleSource"))
1116 );
1117
1118 HoleCpyInfo.DestPosition.X = x * (HoleResolution.X + TexturePackInterval);
1119 HoleCpyInfo.DestPosition.Y = y * (HoleResolution.Y + TexturePackInterval);
1120 HoleCpyInfo.DestPosition.Z = z * (HoleResolution.Z + TexturePackInterval);
1121 AddCopyTexturePass(GraphBuilder, SourceTexture, PackedHoleAtlas, HoleCpyInfo);
1122 }
1123 }
1124 }
1125
1126 // Create GPU buffers
1127 FGlobalShaderMap* ShaderMap = GetGlobalShaderMap(View.FeatureLevel);
1128
1129 FRDGBufferDesc BirthBufferDesc = FRDGBufferDesc::CreateStructuredDesc(sizeof(float), RenderData.PackedVoxelBirthTimes.Num());
1130 FRDGBufferRef BirthBuffer = GraphBuilder.CreateBuffer(BirthBufferDesc, TEXT("IVSmoke_PackedBirthBuffer"));
1131 GraphBuilder.QueueBufferUpload(BirthBuffer, RenderData.PackedVoxelBirthTimes.GetData(), RenderData.PackedVoxelBirthTimes.Num() * sizeof(float));
1132
1133 FRDGBufferDesc DeathBufferDesc = FRDGBufferDesc::CreateStructuredDesc(sizeof(float), RenderData.PackedVoxelDeathTimes.Num());
1134 FRDGBufferRef DeathBuffer = GraphBuilder.CreateBuffer(DeathBufferDesc, TEXT("IVSmoke_PackedDeathBuffer"));
1135 GraphBuilder.QueueBufferUpload(DeathBuffer, RenderData.PackedVoxelDeathTimes.GetData(), RenderData.PackedVoxelDeathTimes.Num() * sizeof(float));
1136
1137 FRDGBufferDesc VolumeBufferDesc = FRDGBufferDesc::CreateStructuredDesc(sizeof(FIVSmokeVolumeGPUData), RenderData.VolumeDataArray.Num());
1138 FRDGBufferRef VolumeBuffer = GraphBuilder.CreateBuffer(VolumeBufferDesc, TEXT("IVSmokeVolumeDataBuffer"));
1139 GraphBuilder.QueueBufferUpload(VolumeBuffer, RenderData.VolumeDataArray.GetData(), RenderData.VolumeDataArray.Num() * sizeof(FIVSmokeVolumeGPUData));
1140
1141 // StructuredToTexture Pass
1142 TShaderMapRef<FIVSmokeStructuredToTextureCS> StructuredCopyShader(ShaderMap);
1143 auto* StructuredCopyParams = GraphBuilder.AllocParameters<FIVSmokeStructuredToTextureCS::FParameters>();
1144 StructuredCopyParams->Desti = GraphBuilder.CreateUAV(PackedVoxelAtlas);
1145 StructuredCopyParams->BirthTimes = GraphBuilder.CreateSRV(BirthBuffer);
1146 StructuredCopyParams->DeathTimes = GraphBuilder.CreateSRV(DeathBuffer);
1147 StructuredCopyParams->VolumeDataBuffer = GraphBuilder.CreateSRV(VolumeBuffer);
1148 StructuredCopyParams->TexSize = VoxelAtlasResolution;
1149 StructuredCopyParams->VoxelResolution = RenderData.VoxelResolution;
1150 StructuredCopyParams->PackedInterval = TexturePackInterval;
1151 StructuredCopyParams->GameTime = RenderData.GameTime;
1152
1153 FIVSmokePostProcessPass::AddComputeShaderPass<FIVSmokeStructuredToTextureCS>(
1154 GraphBuilder,
1155 ShaderMap,
1156 StructuredCopyShader,
1157 StructuredCopyParams,
1158 VoxelAtlasResolution
1159 );
1160
1161 // Voxel FXAA Pass
1162 TShaderMapRef<FIVSmokeVoxelFXAACS> VoxelFXAAShader(ShaderMap);
1163 auto* VoxelFXAAParams = GraphBuilder.AllocParameters<FIVSmokeVoxelFXAACS::FParameters>();
1164
1165 VoxelFXAAParams->Desti = GraphBuilder.CreateUAV(PackedVoxelAtlasFXAA);
1166 VoxelFXAAParams->Source = GraphBuilder.CreateSRV(PackedVoxelAtlas);
1167 VoxelFXAAParams->LinearBorder_Sampler = TStaticSamplerState<SF_Bilinear, AM_Border, AM_Border, AM_Border>::GetRHI();
1168 VoxelFXAAParams->TexSize = VoxelAtlasFXAAResolution;
1169 VoxelFXAAParams->FXAASpanMax = Settings->FXAASpanMax;
1170 VoxelFXAAParams->FXAARange = Settings->FXAARange;
1171 VoxelFXAAParams->FXAASharpness = Settings->FXAASharpness;
1172
1173 FIVSmokePostProcessPass::AddComputeShaderPass<FIVSmokeVoxelFXAACS>(
1174 GraphBuilder,
1175 ShaderMap,
1176 VoxelFXAAShader,
1177 VoxelFXAAParams,
1178 VoxelAtlasFXAAResolution
1179 );
1180
1181 //~==========================================================================
1182 // Phase 1: Create Occupancy Resources
1183
1184 const FIntPoint TileCount = IVSmokeOccupancy::ComputeTileCount(ViewportSize);
1185 const uint32 StepSliceCount = IVSmokeOccupancy::ComputeStepSliceCount(RenderData.MaxSteps);
1186
1187 FIVSmokeOccupancyResources OccResources = IVSmokeOccupancy::CreateOccupancyResources(
1188 GraphBuilder,
1189 TileCount,
1190 StepSliceCount
1191 );

1193 // Calculate max ray distance and GlobalAABB based on volumes
1194 float MaxRayDistance = 0.0f;
1195 FVector3f GlobalAABBMin(1e10f, 1e10f, 1e10f);
1196 FVector3f GlobalAABBMax(-1e10f, -1e10f, -1e10f);
1197 for (const FIVSmokeVolumeGPUData& VolData : RenderData.VolumeDataArray)
1198 {
1199 FVector3f Extent = VolData.VolumeWorldAABBMax - VolData.VolumeWorldAABBMin;
1200 MaxRayDistance = FMath::Max(MaxRayDistance, Extent.Size());
1201
1202 // Accumulate GlobalAABB
1203 GlobalAABBMin = FVector3f::Min(GlobalAABBMin, VolData.VolumeWorldAABBMin);
1204 GlobalAABBMax = FVector3f::Max(GlobalAABBMax, VolData.VolumeWorldAABBMax);
1205 }
1206 MaxRayDistance = FMath::Max(MaxRayDistance, 10000.0f); // Minimum reasonable distance
1207
1208 // MinStepSize from settings (minimum world units per step, TotalVolumeLength computed per-tile in shader)

1211 //~==========================================================================
1212 // Phase 2: Pass 0 - Tile Setup

1214 IVSmokeOccupancy::AddTileSetupPass(
1215 GraphBuilder,
1216 View,
1217 VolumeBuffer,
1218 RenderData.VolumeDataArray.Num(),
