73bool UIVSmokeSettings::IsSelfShadowingEnabled() const
74{
75 if (GlobalQuality == EIVSmokeGlobalQuality::Custom)
76 {
77 return SelfShadowQuality != EIVSmokeSelfShadowQuality::Off;
78 }
79 // Global Low/Medium/High → Self Shadow Low/Medium/High (always enabled)
