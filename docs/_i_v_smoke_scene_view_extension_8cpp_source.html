31void FIVSmokeSceneViewExtension::BeginRenderViewFamily(FSceneViewFamily& InViewFamily)
32{
33 // Called ONCE per frame on Game Thread (not per-view!)
34 // This ensures render data is prepared exactly once per frame
35 FIVSmokeRenderer& Renderer = FIVSmokeRenderer::Get();
36
37 // Skip if no volumes
38 if (!Renderer.HasVolumes())
39 {
40 return;
41 }
42
43 // Collect valid volumes under lock
44 TArray<AIVSmokeVoxelVolume*> ValidVolumes;
45 {
46 FScopeLock Lock(&Renderer.GetVolumesMutex());
47 for (const auto& WeakVolume : Renderer.GetVolumes())
48 {
49 if (AIVSmokeVoxelVolume* Volume = WeakVolume.Get())
50 {
51 ValidVolumes.Add(Volume);
52 }
53 }
54 }
55
56 if (ValidVolumes.Num() == 0)
57 {
58 return;
59 }
60
61 // Prepare render data on Game Thread (all Volume data access happens here)
62 FIVSmokePackedRenderData RenderData = Renderer.PrepareRenderData(ValidVolumes);
63
64 // Transfer to Render Thread via command queue
65 ENQUEUE_RENDER_COMMAND(IVSmokeSetRenderData)(
66 [&Renderer, RenderData = MoveTemp(RenderData)](FRHICommandListImmediate& RHICmdList) mutable
67 {
68 Renderer.SetCachedRenderData(MoveTemp(RenderData));
69 }
70 );
71}

78void FIVSmokeSceneViewExtension::SubscribeToPostProcessingPass(
79 EPostProcessingPass Pass,
80 const FSceneView& InView,
81 FPostProcessingPassDelegateArray& InOutPassCallbacks,
82 bool bIsPassEnabled)
83{
84 // Map IVSmoke render pass setting to engine post-processing pass
85 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
86 EIVSmokeRenderPass RenderPassSetting = Settings ? Settings->RenderPass : EIVSmokeRenderPass::TranslucencyAfterDOF;
87
88 EPostProcessingPass TargetPass;
89 switch (RenderPassSetting)
90 {
91 case EIVSmokeRenderPass::BeforeDOF:
92 TargetPass = EPostProcessingPass::BeforeDOF;
93 break;
94 case EIVSmokeRenderPass::AfterDOF:
95 TargetPass = EPostProcessingPass::AfterDOF;
96 break;
97 case EIVSmokeRenderPass::TranslucencyAfterDOF:
98 TargetPass = EPostProcessingPass::TranslucencyAfterDOF;
99 break;
100 case EIVSmokeRenderPass::MotionBlur:
101 TargetPass = EPostProcessingPass::MotionBlur;
102 break;
103 case EIVSmokeRenderPass::Tonemap:
104 TargetPass = EPostProcessingPass::Tonemap;
105 break;
106 default:
107 TargetPass = EPostProcessingPass::TranslucencyAfterDOF;
108 break;
109 }
110
111 if (Pass == TargetPass)
112 {
113 InOutPassCallbacks.Add(
114 FPostProcessingPassDelegate::CreateRaw(
115 this,
116 &FIVSmokeSceneViewExtension::Render_RenderThread
117 )
118 );
119 }
120}
