33void FIVSmokeSceneViewExtension::BeginRenderViewFamily(FSceneViewFamily& InViewFamily)
34{
35 // Called ONCE per frame on Game Thread (not per-view!)
36 // This ensures render data is prepared exactly once per frame
37 FIVSmokeRenderer& Renderer = FIVSmokeRenderer::Get();
38
39 // Get world from ViewFamily
40 UWorld* World = nullptr;
41 if (InViewFamily.Scene)
42 {
43 World = InViewFamily.Scene->GetWorld();
44 }
45
46 if (!World)
47 {
48 return;
49 }
50
51 // Sync server time if needed
52 if (!Renderer.bIsServerTimeSynced())
53 {
54 if (AGameStateBase* GS = World->GetGameState())
55 {
56 float LocalTime = World->GetTimeSeconds();
57 float ServerTime = GS->GetServerWorldTimeSeconds();
58 Renderer.SetServerTimeOffset(ServerTime - LocalTime);
59 }
60 }
61
62 // Collect renderable volumes using TActorIterator (Pull-based pattern)
63 TArray<AIVSmokeVoxelVolume*> ValidVolumes;
64 for (TActorIterator<AIVSmokeVoxelVolume> It(World); It; ++It)
65 {
66 if (It->ShouldRender())
67 {
68 ValidVolumes.Add(*It);
69 }
70 }
71
72 if (ValidVolumes.Num() == 0)
73 {
74 // Clear cached render data to stop rendering
75 ENQUEUE_RENDER_COMMAND(IVSmokeClearRenderData)(
76 [&Renderer](FRHICommandListImmediate& RHICmdList)
77 {
78 Renderer.SetCachedRenderData(FIVSmokePackedRenderData());
79 }
80 );
81 return;
82 }
83
84 // Get camera position from first view for distance-based filtering
85 FVector CameraPosition = FVector::ZeroVector;
86 if (InViewFamily.Views.Num() > 0 && InViewFamily.Views[0])
87 {
88 CameraPosition = InViewFamily.Views[0]->ViewLocation;
89 }
90
91 // Prepare render data on Game Thread (all Volume data access happens here)
92 FIVSmokePackedRenderData RenderData = Renderer.PrepareRenderData(ValidVolumes, CameraPosition);
93
94 // Transfer to Render Thread via command queue
95 ENQUEUE_RENDER_COMMAND(IVSmokeSetRenderData)(
96 [&Renderer, RenderData = MoveTemp(RenderData)](FRHICommandListImmediate& RHICmdList) mutable
97 {
98 Renderer.SetCachedRenderData(MoveTemp(RenderData));
99 }
100 );
101}

110void FIVSmokeSceneViewExtension::SubscribeToPostProcessingPass(
111 EPostProcessingPass Pass,
112 const FSceneView& InView,
113 FPostProcessingPassDelegateArray& InOutPassCallbacks,
114 bool bIsPassEnabled)
115{
116 // Map IVSmoke render pass setting to engine post-processing pass
117 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
118 EIVSmokeRenderPass RenderPassSetting = Settings ? Settings->RenderPass : EIVSmokeRenderPass::TranslucencyAfterDOF;
119
120 EPostProcessingPass TargetPass;
121 switch (RenderPassSetting)
122 {
123 case EIVSmokeRenderPass::BeforeDOF:
124 TargetPass = EPostProcessingPass::BeforeDOF;
125 break;
126 case EIVSmokeRenderPass::AfterDOF:
127 TargetPass = EPostProcessingPass::AfterDOF;
128 break;
129 case EIVSmokeRenderPass::TranslucencyAfterDOF:
130 TargetPass = EPostProcessingPass::TranslucencyAfterDOF;
131 break;
132 case EIVSmokeRenderPass::MotionBlur:
133 TargetPass = EPostProcessingPass::MotionBlur;
134 break;
135 case EIVSmokeRenderPass::Tonemap:
136 TargetPass = EPostProcessingPass::Tonemap;
137 break;
138 default:
139 TargetPass = EPostProcessingPass::TranslucencyAfterDOF;
140 break;
141 }
142
143 if (Pass == TargetPass)
144 {
145 InOutPassCallbacks.Add(
146 FPostProcessingPassDelegate::CreateRaw(
147 this,
148 &FIVSmokeSceneViewExtension::Render_RenderThread
149 )
150 );
151 }
152}
