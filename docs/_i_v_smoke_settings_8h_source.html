10/**
11 * Post-processing pass where smoke is rendered.
12 * Affects interaction with particles, DOF, Bloom, and other effects.
13 */
14UENUM(BlueprintType)
15enum class EIVSmokeRenderPass : uint8
16{
17 /** Before Depth of Field. Best quality but particles may render on top. */
18 BeforeDOF UMETA(DisplayName = "Before DOF (Best Quality)"),
19
20 /** After Depth of Field. DOF applied to smoke. Recommended for most cases. */
21 AfterDOF UMETA(DisplayName = "After DOF (Recommended)"),
22
23 /** Translucency After DOF. Smoke renders over AfterDOF particles. Experimental. */
24 TranslucencyAfterDOF UMETA(DisplayName = "Translucency After DOF (Experimental)"),
25
26 /** After Motion Blur. Most effects applied but may cause edge artifacts.
27 * @deprecated Not recommended due to visual artifacts. */
28 MotionBlur UMETA(DisplayName = "After Motion Blur", Hidden),
29
30 /** After Tonemapping. All particles rendered below, but no Bloom/DOF/TAA on smoke.
31 * @deprecated Not recommended due to missing post-processing effects. */
32 Tonemap UMETA(DisplayName = "After Tonemap (No Post Effects)", Hidden)
33};
