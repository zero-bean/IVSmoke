51UENUM(BlueprintType)
52enum class EIVSmokeRenderPass : uint8
53{
54 /** Before Depth of Field. Best quality but particles may render on top. */
55 BeforeDOF UMETA(DisplayName = "Before DOF (Best Quality)"),
56
57 /** After Depth of Field. DOF applied to smoke. Recommended for most cases. */
58 AfterDOF UMETA(DisplayName = "After DOF (Recommended)"),
59
60 /** Translucency After DOF. Smoke renders over AfterDOF particles. Experimental. */
61 TranslucencyAfterDOF UMETA(DisplayName = "Translucency After DOF (Experimental)"),
62
63 /** After Motion Blur. Most effects applied but may cause edge artifacts. */
64 MotionBlur UMETA(DisplayName = "After Motion Blur"),
65
66 /** After Tonemapping. All particles rendered below, but no Bloom/DOF/TAA on smoke. */
67 Tonemap UMETA(DisplayName = "After Tonemap (No Post Effects)")
68};
