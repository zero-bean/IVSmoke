139/**
140 * The core volumetric actor that simulates dynamic smoke expansion using a deterministic voxel-based flood-fill algorithm.
141 *
142 * ## Overview
143 * This actor generates a 3D grid of voxels that expand outward from the center, navigating around obstacles
144 * defined by the collision settings. The simulation is*deterministic, ensuring the same shape and timing
145 * across both Server and Clients without replicating individual voxel data.
146 *
147 * ## Simulation Lifecycle
148 * The simulation state machine progresses based on the sum of duration and fade settings:
149 * 1. **Idle**: Initial state.
150 * 2. **Expansion**: Spawns voxels. Ends after `ExpansionDuration + FadeInDuration`.
151 * 3. **Sustain**: Maintains the shape. Ends after `SustainDuration`.
152 * 4. **Dissipation**: Removes voxels. Ends after `DissipationDuration + FadeOutDuration`.
153 * 5. **Finished**: Simulation complete.
154 *
155 * ## Network & Execution
156 * The simulation logic executes deterministically on both the Server and Client.
157 * - The Server manages the authoritative state (State, Seed, StartTime) and replicates it to Clients.
158 * - Clients execute the exact same flood-fill algorithm locally based on the replicated Seed and Time.
159 */

397 /** Internal node structure for the Dijkstra-based flood fill algorithm. */
398 struct FIVSmokeVoxelNode
399 {
400 int32 Index;
401 int32 ParentIndex;
402 float Cost;
403 bool operator<(const FIVSmokeVoxelNode& Other) const
404 {
405 if (FMath::IsNearlyEqual(Cost, Other.Cost))
406 {
407 return Index < Other.Index;
408 }
409 return Cost < Other.Cost;
410 }
411 };
412
413 /**
414 * Helper to sample a curve or return linear alpha if no curve is provided.
415 *
416 * @param ElapsedTime Current time elapsed in the phase.
417 * @param Duration Total duration of the phase.
418 * @param Curve Optional curve to sample. If null, returns linear interpolation (ElapsedTime / Duration).
419 * @return Clamped float value between 0.0 and 1.0.
420 */
421 FORCEINLINE static float GetCurveValue(float ElapsedTime, float Duration, const UCurveFloat* Curve)
422 {
423 if (Duration <= KINDA_SMALL_NUMBER)
424 {
425 return 1.0f;
426 }
427
428 float Alpha = FMath::Clamp(ElapsedTime / Duration, 0.0f, 1.0f);
429
430 if (Curve)
431 {
432 return FMath::Clamp(Curve->GetFloatValue(Alpha), 0.0f, 1.0f);
433 }
434
435 return Alpha;
436 }
