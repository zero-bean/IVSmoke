152void UIVSmokeHoleGeneratorComponent::Internal_RequestPenetrationHole_Implementation(const FVector3f& InOrigin, const FVector3f& InDirection, const uint8 PresetID)
153{
154 const TObjectPtr<UIVSmokeHolePreset> Preset = UIVSmokeHolePreset::FindByID(PresetID);
155 if (!Preset)
156 {
157 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestPenetrationHole] Invalid PresetID: %d"), PresetID);
158 return;
159 }
160
161 if (Preset->Duration <= 0.0f)
162 {
163 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestPenetrationHole] Invalid Lifetime: %f"), Preset->Duration);
164 return;
165 }
166
167 if (Preset->HoleType != EIVSmokeHoleType::Penetration)
168 {
169 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestPenetrationHole] Preset is not Penetration type"));
170 return;
171 }
172
173 // 1. Check whether it passes through the smoke volume
174 FVector3f EntryPoint, ExitPoint;
175 if (!Authority_CalculatePenetrationPoints(InOrigin, InDirection, Preset->Radius, EntryPoint, ExitPoint))
176 {
177 return;
178 }
179
180 // 2. Create Hole
181 FIVSmokeHoleData HoleData;
182 HoleData.Position = EntryPoint;
183 HoleData.EndPosition = ExitPoint;
184 HoleData.PresetID = PresetID;
185 HoleData.ExpirationServerTime = GetSyncedTime() + Preset->Duration;
186 Authority_CreateHole(HoleData);
187}

189void UIVSmokeHoleGeneratorComponent::Internal_RequestExplosionHole_Implementation(const FVector3f& Origin, const uint8 PresetID)
190{
191 const TObjectPtr<UIVSmokeHolePreset> Preset = UIVSmokeHolePreset::FindByID(PresetID);
192 if (!Preset)
193 {
194 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestExplosionHole] Invalid PresetID: %d"), PresetID);
195 return;
196 }
197
198 if (Preset->Duration <= 0.0f)
199 {
200 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestExplosionHole] Invalid Lifetime: %f"), Preset->Duration);
201 return;
202 }
203
204 if (Preset->HoleType != EIVSmokeHoleType::Explosion)
205 {
206 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestExplosionHole] Preset is not Explosion type"));
207 return;
208 }
209
210 // 1. Check whether the smoke volume intersects with the explosion
211 const FBox3f VolumeBox = FBox3f(Bounds.GetBox());
212 const FVector3f ExpandedMin = VolumeBox.Min - FVector3f(Preset->Radius);
213 const FVector3f ExpandedMax = VolumeBox.Max + FVector3f(Preset->Radius);
214 if (const FBox3f ExpandedBox(ExpandedMin, ExpandedMax); !ExpandedBox.IsInside(Origin))
215 {
216 return;
217 }
218
219 // 2. Create Hole
220 FIVSmokeHoleData HoleData;
221 HoleData.Position = Origin;
222 HoleData.EndPosition = Origin;
223 HoleData.PresetID = PresetID;
224 HoleData.ExpirationServerTime = GetSyncedTime() + Preset->Duration;
225 Authority_CreateHole(HoleData);
226}

228void UIVSmokeHoleGeneratorComponent::Internal_RequestDynamicHole_Implementation(AActor* TargetActor, const uint8 PresetID)
229{
230 const TObjectPtr<UIVSmokeHolePreset> Preset = UIVSmokeHolePreset::FindByID(PresetID);
231 if (!Preset)
232 {
233 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestDynamicHole] Invalid PresetID: %d"), PresetID);
234 return;
235 }
236
237 if (Preset->Duration <= 0.0f)
238 {
239 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestDynamicHole] Invalid Duration: %f"), Preset->Duration);

241 }
242
243 if (Preset->HoleType != EIVSmokeHoleType::Dynamic)
244 {
245 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestDynamicHole] Preset is not Dynamic type"));
246 return;
247 }
248
249 // 1. Check if already registered
250 for (const FIVSmokeHoleDynamicSubject& Tracker : DynamicSubjectList)
251 {
252 if (Tracker.TargetActor == TargetActor)
253 {
254 UE_LOG(LogIVSmoke, Warning, TEXT("[Internal_RequestDynamicHole] Actor already registered"));
255 return;
256 }
257 }
258
259 // 2. Register new subject
260 FIVSmokeHoleDynamicSubject NewDynamicSubject;
261 NewDynamicSubject.TargetActor = TargetActor;
262 NewDynamicSubject.PresetID = PresetID;
263 NewDynamicSubject.LastWorldPosition = FVector3f(TargetActor->GetActorLocation());
264 NewDynamicSubject.LastWorldRotation = TargetActor->GetActorQuat();
265 DynamicSubjectList.Add(NewDynamicSubject);
266}

272void UIVSmokeHoleGeneratorComponent::Authority_CreateHole(const FIVSmokeHoleData& HoleData)
273{
274 if (ActiveHoles.Num() < MaxHoles)
275 {
276 ActiveHoles.AddHole(HoleData);
277 }
278 else
279 {
280 int32 OldestIndex = 0;
281 float MinTime = FLT_MAX;
282
283 for (int32 i = 0; i < ActiveHoles.Num(); ++i)
284 {
285 if (ActiveHoles[i].ExpirationServerTime < MinTime)
286 {
287 MinTime = ActiveHoles[i].ExpirationServerTime;
288 OldestIndex = i;
289 }
290 }
291
292 FIVSmokeHoleData& Target = ActiveHoles[OldestIndex];
293 Target.Position = HoleData.Position;
294 Target.EndPosition = HoleData.EndPosition;
295 Target.ExpirationServerTime = HoleData.ExpirationServerTime;
296 Target.PresetID = HoleData.PresetID;
297 ActiveHoles.MarkItemDirty(Target);
298 }
299
300 MarkHoleTextureDirty();
301}

317bool UIVSmokeHoleGeneratorComponent::Authority_CalculatePenetrationPoints(
318 const FVector3f& Origin, const FVector3f& Direction, const float Radius, FVector3f& OutEntry, FVector3f& OutExit)
319{
320 FVector3f NormalizedDirection = Direction.GetSafeNormal();
321 if (NormalizedDirection.IsNearlyZero())
322 {
323 UE_LOG(LogIVSmoke, Warning, TEXT("[CalculatePenetrationPoints] Direction is zero"));
324 return false;
325 }
326
327 const float DistToCenter = FVector3f::Dist(Origin, FVector3f(GetComponentLocation()));
328 const float DiagonalLength = GetScaledBoxExtent().Size() * 2.0f;
329 const float MaxDistance = DistToCenter + DiagonalLength;
330
331 const FVector3f RayEnd = Origin + NormalizedDirection * MaxDistance;
332
333 FHitResult HitEntry, HitExit;
334 FCollisionQueryParams QueryParams;
335 QueryParams.bTraceComplex = false;
336
337 // 1. Forward trace (Origin -> RayEnd) to find Entry point
338 if (!LineTraceComponent(HitEntry, FVector(Origin), FVector(RayEnd), QueryParams))
339 {
340 return false;
341 }
342
343 // 2. Reverse trace (RayEnd -> Origin) to find Exit point
344 if (!LineTraceComponent(HitExit, FVector(RayEnd), FVector(Origin), QueryParams))
345 {
346 OutExit = FVector3f(HitEntry.Location);
347 }
348 else
349 {
350 OutExit = FVector3f(HitExit.Location);
351 }
352
353 OutEntry = FVector3f(HitEntry.Location);
354
355 // 3. Obstacle detection using SphereTrace between Entry and Exit
356 if (ObstacleObjectTypes.Num() > 0)
357 {
358 FHitResult ObstacleHit;
359 FCollisionQueryParams WorldParams;
360 const FCollisionShape SweepShape = FCollisionShape::MakeSphere(Radius);
361 const FCollisionObjectQueryParams ObjectParams(ObstacleObjectTypes);
362
363 if (GetWorld()->SweepSingleByObjectType(
364 ObstacleHit, FVector(OutEntry), FVector(OutExit), FQuat::Identity,
365 ObjectParams, SweepShape, WorldParams))
366 {
367 OutExit = FVector3f(ObstacleHit.Location);
368 }
369 }
370
371 return true;
372}

374void UIVSmokeHoleGeneratorComponent::Authority_UpdateDynamicSubjectList()
375{
376 const float CurrentTime = GetSyncedTime();
377 const FBox3f SmokeVolume = FBox3f(Bounds.GetBox());
378
379 for (int32 i = DynamicSubjectList.Num() - 1; i >= 0; --i)
380 {
381 FIVSmokeHoleDynamicSubject& DynamicSubject = DynamicSubjectList[i];
382
383 // 0. Delete if object is not alive
384 if (!DynamicSubject.TargetActor.IsValid())
385 {
386 DynamicSubjectList.RemoveAtSwap(i);
387 continue;
388 }
389
390 const TObjectPtr<AActor> Actor = DynamicSubject.TargetActor.Get();
391 const TObjectPtr<UIVSmokeHolePreset> Preset = UIVSmokeHolePreset::FindByID(DynamicSubject.PresetID);
392
393 // 0. Delete if preset is not valid
394 if (!Preset)
395 {
396 DynamicSubjectList.RemoveAtSwap(i);
397 continue;
398 }
399
400 const FVector3f CurrentPos = FVector3f(Actor->GetActorLocation());
401 const FVector3f LastPos = FVector3f(DynamicSubject.LastWorldPosition);
402
403 if (!SmokeVolume.IsInside(CurrentPos))
404 {
405 continue;
406 }
407
408 // 1. Ignore if object moves a little bit
409 if (Preset->DistanceThreshold > FVector3f::Dist(CurrentPos, LastPos))
410 {
411 continue;
412 }
413
414 // 2. Create Hole
415 FIVSmokeHoleData HoleData;
416 HoleData.Position = LastPos;
417 HoleData.EndPosition = CurrentPos;
418 HoleData.PresetID = DynamicSubject.PresetID;
419 HoleData.ExpirationServerTime = CurrentTime + Preset->Duration;
420 Authority_CreateHole(HoleData);
421
422 // 3. Update registered object
423 DynamicSubject.LastWorldPosition = CurrentPos;
424 DynamicSubject.LastWorldRotation = Actor->GetActorQuat();
425 }
426}

449void UIVSmokeHoleGeneratorComponent::Local_RebuildHoleTexture()
450{
451 if (!HoleTexture)
452 {
453 return;
454 }
455
456 const FTextureRenderTargetResource* RenderTargetResource = HoleTexture->GameThread_GetRenderTargetResource();
457 if (!RenderTargetResource)
458 {
459 return;
460 }
461
462 TArray<FIVSmokeHoleGPU> GPUHoles = ActiveHoles.GetHoleGPUData(GetSyncedTime());
463
464 const TObjectPtr<AIVSmokeVoxelVolume> VoxelVolume = Cast<AIVSmokeVoxelVolume>(GetOwner());

471 const FVector3f WorldVolumeMax = FVector3f(VoxelVolume->GetVoxelWorldAABBMax());
472
473 const FIntVector Resolution = VoxelResolution;
474 const int32 NumHoles = ActiveHoles.Num();
475 FTextureRHIRef Texture = RenderTargetResource->GetRenderTargetTexture();
476
477 ENQUEUE_RENDER_COMMAND(IVSmokeHoleCarveFullRebuild)(
478 [Texture, GPUHoles = MoveTemp(GPUHoles), WorldVolumeMin, WorldVolumeMax, Resolution, NumHoles]
479 (FRHICommandListImmediate& RHICmdList)
480 {
481 FRDGBuilder GraphBuilder(RHICmdList);
482
483 const FRDGTextureRef RDGTexture = GraphBuilder.RegisterExternalTexture(
484 CreateRenderTarget(Texture, TEXT("IVSmokeHoleTexture"))
485 );
486
487 const FRDGBufferRef HoleBuffer = CreateStructuredBuffer(
488 GraphBuilder,
489 TEXT("IVSmokeHoleBuffer"),
490 sizeof(FIVSmokeHoleGPU),
491 GPUHoles.Num(),
492 GPUHoles.GetData(),
493 sizeof(FIVSmokeHoleGPU) * GPUHoles.Num()
494 );
495
496 FIVSmokeHoleCarveCS::FParameters* Parameters = GraphBuilder.AllocParameters<FIVSmokeHoleCarveCS::FParameters>();
497 Parameters->VolumeTexture = GraphBuilder.CreateUAV(RDGTexture);
498 Parameters->HoleBuffer = GraphBuilder.CreateSRV(HoleBuffer);
499 Parameters->VolumeMin = WorldVolumeMin;
500 Parameters->VolumeMax = WorldVolumeMax;
501 Parameters->Resolution = Resolution;
502 Parameters->NumHoles = NumHoles;
503
504 const TShaderMapRef<FIVSmokeHoleCarveCS> ComputeShader(GetGlobalShaderMap(GMaxRHIFeatureLevel));
505 FIVSmokePostProcessPass::AddComputeShaderPass<FIVSmokeHoleCarveCS>(GraphBuilder, GetGlobalShaderMap(GMaxRHIFeatureLevel), ComputeShader, Parameters, Resolution);
506 GraphBuilder.Execute();
507 }
508 );
