27void FIVSmokeCSMRenderer::Initialize(UWorld* World, int32 NumCascades, int32 Resolution, float MaxDistance)
28{
29 if (!World)
30 {
31 UE_LOG(LogIVSmokeCSM, Error, TEXT("[FIVSmokeCSMRenderer::Initialize] World is null"));
32 return;
33 }
34
35 // Clean up existing resources
36 Shutdown();
37
38 // Clamp parameters
39 NumCascades = FMath::Clamp(NumCascades, 1, 8);
40 Resolution = FMath::Clamp(Resolution, 256, 2048);
41 MaxDistance = FMath::Max(MaxDistance, 1000.0f);
42
43 CurrentResolution = Resolution;
44 MaxShadowDistance = MaxDistance;
45
46 // Load settings
47 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
48 if (Settings)
49 {
50 LogLinearBlend = Settings->CascadeLogLinearBlend;
51 bEnablePriorityUpdate = Settings->bEnablePriorityUpdate;
52 NearCascadeUpdateInterval = Settings->NearCascadeUpdateInterval;
53 FarCascadeUpdateInterval = Settings->FarCascadeUpdateInterval;
54 }
55
56 // Create owner actor for capture components
57 FActorSpawnParameters SpawnParams;
58 SpawnParams.Name = TEXT("IVSmokeCSMCaptureOwner");
59 SpawnParams.ObjectFlags |= RF_Transient;
60 AActor* Owner = World->SpawnActor<AActor>(AActor::StaticClass(), FVector::ZeroVector, FRotator::ZeroRotator, SpawnParams);
61 if (!Owner)
62 {
63 UE_LOG(LogIVSmokeCSM, Error, TEXT("[FIVSmokeCSMRenderer::Initialize] Failed to create capture owner actor"));
64 return;
65 }
66 CaptureOwner = Owner;
67
68 // Initialize cascades
69 Cascades.SetNum(NumCascades);
70 for (int32 i = 0; i < NumCascades; i++)
71 {
72 FIVSmokeCascadeData& Cascade = Cascades[i];
73 Cascade.CascadeIndex = i;
74
75 // Create capture component
76 USceneCaptureComponent2D* CaptureComp = NewObject<USceneCaptureComponent2D>(Owner);
77 CaptureComp->RegisterComponent();
78 CaptureComp->AttachToComponent(Owner->GetRootComponent(), FAttachmentTransformRules::KeepWorldTransform);
79 ConfigureCaptureComponent(CaptureComp);
80 Cascade.CaptureComponent = CaptureComp;
81
82 // Create render targets
83 CreateCascadeRenderTargets(Cascade, Resolution);
84
85 // Assign depth RT to capture component
86 CaptureComp->TextureTarget = Cascade.DepthRT;
87 }
88
89 // Calculate initial cascade splits
90 CalculateCascadeSplits(NearPlaneDistance, MaxDistance, LogLinearBlend);
91
92 bIsInitialized = true;
93
94 UE_LOG(LogIVSmokeCSM, Log, TEXT("[FIVSmokeCSMRenderer::Initialize] Initialized with %d cascades, %dx%d resolution, %.0f max distance"),
95 NumCascades, Resolution, Resolution, MaxDistance);
96}

98void FIVSmokeCSMRenderer::Shutdown()
99{
100 for (FIVSmokeCascadeData& Cascade : Cascades)
101 {
102 // Clean up render targets
103 if (Cascade.DepthRT)
104 {
105 Cascade.DepthRT->RemoveFromRoot();
106 Cascade.DepthRT = nullptr;
107 }
108 if (Cascade.VSMRT)
109 {
110 Cascade.VSMRT->RemoveFromRoot();
111 Cascade.VSMRT = nullptr;
112 }
113
114 // Capture component will be destroyed with owner
115 Cascade.CaptureComponent = nullptr;
116 }
117
118 Cascades.Empty();
119
120 // Destroy owner actor
121 if (CaptureOwner.IsValid())
122 {
123 CaptureOwner->Destroy();
124 CaptureOwner = nullptr;
125 }
126
127 bIsInitialized = false;
128
129 UE_LOG(LogIVSmokeCSM, Log, TEXT("[FIVSmokeCSMRenderer::Shutdown] CSM renderer shut down"));
130}

135void FIVSmokeCSMRenderer::Update(
136 const FVector& CameraPosition,
137 const FVector& CameraForward,
138 const FVector& LightDirection,
139 uint32 FrameNumber)
140{
141 if (!bIsInitialized || Cascades.Num() == 0)
142 {
143 return;
144 }
145
146 // Store main camera position for camera-relative calculations
147 MainCameraPosition = CameraPosition;
148
149 //~==========================================================================
150 // Validity Check - PIE restart can invalidate capture components
151 // Check CaptureOwner first (all components are attached to it).
152 // If owner is destroyed, all components are invalid.
153 if (!CaptureOwner.IsValid())
154 {
155 UE_LOG(LogIVSmokeCSM, Warning, TEXT("[FIVSmokeCSMRenderer::Update] CaptureOwner invalidated (PIE restart?), shutting down for reinitialization"));
156 Shutdown();
157 return;
158 }
159
160 // Also verify at least one capture component is valid
161 bool bAnyValidComponent = false;
162 for (const FIVSmokeCascadeData& Cascade : Cascades)
163 {
164 if (IsValid(Cascade.CaptureComponent))
165 {
166 bAnyValidComponent = true;
167 break;
168 }
169 }
170 if (!bAnyValidComponent)
171 {
172 UE_LOG(LogIVSmokeCSM, Warning, TEXT("[FIVSmokeCSMRenderer::Update] All CaptureComponents invalidated, shutting down for reinitialization"));
173 Shutdown();
174 return;
175 }
176
177 // Determine which cascades need update
178 UpdateCascadePriorities(FrameNumber);
179
180 // Update each cascade that needs it
181 for (int32 i = 0; i < Cascades.Num(); i++)
182 {
183 if (Cascades[i].bNeedsCapture)
184 {
185 UpdateCascadeCapture(i, CameraPosition, LightDirection);
186 Cascades[i].LastCaptureFrame = FrameNumber;
187 }
188 }
189}

191void FIVSmokeCSMRenderer::UpdateCascadePriorities(uint32 FrameNumber)
192{
193 if (!bEnablePriorityUpdate)
194 {
195 // Update all cascades every frame
196 for (FIVSmokeCascadeData& Cascade : Cascades)
197 {
198 Cascade.bNeedsCapture = true;
199 }
200 return;
201 }
202
203 const int32 NumCascades = Cascades.Num();
204 const int32 MidCascade = NumCascades / 2;
205
206 for (int32 i = 0; i < NumCascades; i++)
207 {
208 int32 UpdateInterval;
209
210 if (i < MidCascade)
211 {
212 // Near cascades: update frequently
213 UpdateInterval = NearCascadeUpdateInterval;
214 }
215 else
216 {
217 // Far cascades: update less frequently
218 // Interpolate between near and far intervals based on cascade position
219 float t = (float)(i - MidCascade) / (float)(NumCascades - MidCascade);
220 UpdateInterval = FMath::RoundToInt(FMath::Lerp((float)NearCascadeUpdateInterval, (float)FarCascadeUpdateInterval, t));
221 }
222
223 Cascades[i].bNeedsCapture = (FrameNumber % UpdateInterval == 0);
224 }
225}

227void FIVSmokeCSMRenderer::UpdateCascadeCapture(
228 int32 CascadeIndex,
229 const FVector& CameraPosition,
230 const FVector& LightDirection)
231{
232 if (!Cascades.IsValidIndex(CascadeIndex))
233 {
234 return;
235 }
236
237 FIVSmokeCascadeData& Cascade = Cascades[CascadeIndex];
238 if (!IsValid(Cascade.CaptureComponent))
239 {
240 UE_LOG(LogIVSmokeCSM, Warning, TEXT("[FIVSmokeCSMRenderer::UpdateCascadeCapture] CaptureComponent for cascade %d is invalid"), CascadeIndex);
241 return;
242 }
243
244 //~==========================================================================
245 // SINGLE-BUFFER TIMING MODEL
246 //
247 // SceneCaptureComponent2D with bCaptureEveryFrame=true captures during the
248 // SAME frame's render pass. Therefore:
249 // 1. We calculate VP matrix and set CaptureComponent transform here
250 // 2. The capture executes during this frame's render pass
251 // 3. The ray march shader samples the captured texture with our VP matrix
252 //
253 // All operations use the SAME values in the SAME frame - no buffering needed.
254
255 // Normalize light direction
256 FVector NormalizedLightDir = LightDirection.GetSafeNormal();
257 if (NormalizedLightDir.IsNearlyZero())
258 {
259 NormalizedLightDir = FVector(0.0f, 0.0f, 1.0f);
260 }
261
262 //~==========================================================================
263 // CSM Camera Positioning
264 //
265 // OrthoWidth covers the cascade's view frustum at its far distance.
266 // For a reasonable FOV (~90Â°), width at distance D is roughly 2*D.
267 float NewOrthoWidth = Cascade.FarPlane * 2.0f;
268
269 // Calculate light view axes
270 FVector LightForward = -NormalizedLightDir; // Camera looks opposite to light direction
271 FVector LightRight = FVector::CrossProduct(NormalizedLightDir, FVector::UpVector);
272 if (LightRight.IsNearlyZero())
273 {
274 LightRight = FVector::CrossProduct(NormalizedLightDir, FVector::ForwardVector);
275 }
276 LightRight.Normalize();
277 FVector LightUp = FVector::CrossProduct(LightRight, LightForward).GetSafeNormal();
278
279 // Position shadow camera far enough to see all shadow casters
280 float CaptureDistance = MaxShadowDistance * 1.5f;
281 FVector BaseCapturePosition = CameraPosition + NormalizedLightDir * CaptureDistance;
282
283 //~==========================================================================
284 // Texel Snapping - Use SMALLEST cascade's texel size for ALL cascades
285 //
286 // CRITICAL: All cascades must snap to the SAME grid to ensure:
287 // 1. Same WorldPos maps to same relative UV across cascades
288 // 2. Minimal shadow shimmer during camera movement
289 // 3. Smooth cascade transitions without edge artifacts
290 //
291 // Uses cascade 0's texel size (smallest = finest grid) for consistency.
292 double SmallestOrthoWidth = (double)Cascades[0].FarPlane * 2.0;
293 double TexelSize = SmallestOrthoWidth / (double)CurrentResolution;
294
295 // Project PLAYER CAMERA position onto light view axes
296 double PlayerRightOffset = FVector::DotProduct(CameraPosition, LightRight);
297 double PlayerUpOffset = FVector::DotProduct(CameraPosition, LightUp);
298
299 // Snap to unified grid (same for all cascades)
300 double SnappedRight = FMath::FloorToDouble(PlayerRightOffset / TexelSize) * TexelSize;
301 double SnappedUp = FMath::FloorToDouble(PlayerUpOffset / TexelSize) * TexelSize;
302
303 // Calculate adjustment
304 FVector SnapAdjustment = LightRight * (float)(SnappedRight - PlayerRightOffset)
305 + LightUp * (float)(SnappedUp - PlayerUpOffset);
306
307 FVector SnappedPosition = BaseCapturePosition + SnapAdjustment;
308
309 //~==========================================================================
310 // Store current frame values (used by both capture and shader)
311 Cascade.OrthoWidth = NewOrthoWidth;
312 Cascade.LightCameraPosition = SnappedPosition;
313 Cascade.LightCameraForward = LightForward;
314
315 // Apply to capture component
316 FRotator CaptureRotation = LightForward.Rotation();
317 Cascade.CaptureComponent->SetWorldLocationAndRotation(SnappedPosition, CaptureRotation);
318 Cascade.CaptureComponent->OrthoWidth = NewOrthoWidth;
319
320 // Calculate view-projection matrix
321 CalculateViewProjectionMatrix(Cascade);
322
323 UE_LOG(LogIVSmokeCSM, Verbose, TEXT("[FIVSmokeCSMRenderer::UpdateCascadeCapture] Cascade %d: Near=%.0f, Far=%.0f, OrthoWidth=%.0f"),
324 CascadeIndex, Cascade.NearPlane, Cascade.FarPlane, NewOrthoWidth);
325}

330void FIVSmokeCSMRenderer::CalculateCascadeSplits(float NearPlane, float FarPlane, float LogLinearBlendFactor)
331{
332 const int32 NumCascades = Cascades.Num();
333 if (NumCascades == 0)
334 {
335 return;
336 }
337
338 // Ensure minimum near plane
339 NearPlane = FMath::Max(NearPlane, 1.0f);
340
341 for (int32 i = 0; i < NumCascades; i++)
342 {
343 float t = (float)(i + 1) / (float)NumCascades;
344
345 // Linear distribution
346 float Linear = NearPlane + (FarPlane - NearPlane) * t;
347
348 // Logarithmic distribution (better near/far balance)
349 float Log = NearPlane * FMath::Pow(FarPlane / NearPlane, t);
350
351 // Blend between linear and logarithmic
352 float SplitDistance = FMath::Lerp(Linear, Log, LogLinearBlendFactor);
353
354 Cascades[i].FarPlane = SplitDistance;
355 Cascades[i].NearPlane = (i == 0) ? NearPlane : Cascades[i - 1].FarPlane;
356 }
357
358 UE_LOG(LogIVSmokeCSM, Log, TEXT("[FIVSmokeCSMRenderer::CalculateCascadeSplits] Splits calculated (LogLinear=%.2f):"), LogLinearBlendFactor);
359 for (int32 i = 0; i < NumCascades; i++)
360 {
361 UE_LOG(LogIVSmokeCSM, Log, TEXT(" Cascade %d: %.0f - %.0f cm"), i, Cascades[i].NearPlane, Cascades[i].FarPlane);
362 }
363}

367
368FVector FIVSmokeCSMRenderer::ApplyTexelSnapping(
369 const FVector& LightViewOrigin,
370 float OrthoWidth,
371 int32 Resolution,
372 const FVector& LightRight,
373 const FVector& LightUp) const
374{
375 // Calculate world-space texel size
376 // Use double precision for large coordinates to avoid float precision issues
377 double TexelSize = (double)OrthoWidth / (double)Resolution;
378
379 // Project position onto light view axes using double precision
380 // This is critical when coordinates are large (e.g., 100,000 cm)
381 double RightOffset = FVector::DotProduct(LightViewOrigin, LightRight);
382 double UpOffset = FVector::DotProduct(LightViewOrigin, LightUp);
383
384 // Snap to texel grid using Floor (not Round) for stability
385 // Round can cause jittering when value oscillates around X.5
386 double SnappedRight = FMath::FloorToDouble(RightOffset / TexelSize) * TexelSize;
387 double SnappedUp = FMath::FloorToDouble(UpOffset / TexelSize) * TexelSize;
388
389 // Calculate adjustment delta
390 double DeltaRight = SnappedRight - RightOffset;
391 double DeltaUp = SnappedUp - UpOffset;
392
393 // Apply snapped offset
394 FVector Snapped = LightViewOrigin;
395 Snapped += LightRight * (float)DeltaRight;
396 Snapped += LightUp * (float)DeltaUp;
397
398 return Snapped;
399}

402// View-Projection Matrix
403
404void FIVSmokeCSMRenderer::CalculateViewProjectionMatrix(FIVSmokeCascadeData& Cascade)
405{
406 //~==========================================================================
407 // Calculate VP Matrix to MATCH SceneCaptureComponent2D's actual rendering
408 //
409 // We must use the EXACT same method that UE uses for SceneCaptureComponent2D
410 // to ensure our VP matrix matches the captured texture.
411 //
412 // This is read from CaptureComponent to ensure we use the exact same values
413 // that the capture will use (after any engine-side adjustments).
414
415 if (!IsValid(Cascade.CaptureComponent))
416 {
417 return;
418 }
419
420 // Get the actual transform that will be used for capture
421 FVector CameraLocation = Cascade.CaptureComponent->GetComponentLocation();
422 FRotator CameraRotation = Cascade.CaptureComponent->GetComponentRotation();
423
424 //~==========================================================================
425 // View Matrix - Match UE's FSceneView calculation
426 //
427 // UE calculates view matrix from component transform using:
428 // 1. Translation matrix (negative location)
429 // 2. Rotation matrix (inverse rotation + axis swap for UE coordinate system)
430 FMatrix ViewRotationMatrix = FInverseRotationMatrix(CameraRotation) * FMatrix(
431 FPlane(0, 0, 1, 0),
432 FPlane(1, 0, 0, 0),
433 FPlane(0, 1, 0, 0),
434 FPlane(0, 0, 0, 1)
435 );
436 FMatrix ViewMatrix = FTranslationMatrix(-CameraLocation) * ViewRotationMatrix;
437
438 //~==========================================================================
439 // Projection Matrix - Orthographic
440 float OrthoWidth = Cascade.CaptureComponent->OrthoWidth;
441 float HalfWidth = OrthoWidth * 0.5f;
442 float HalfHeight = HalfWidth; // Square projection
443
444 // Near/far planes - use large range to capture all shadow casters
445 float NearZ = 1.0f;
446 float FarZ = MaxShadowDistance * 3.0f;
447
448 // Build orthographic projection matrix (UE uses reversed-Z)
449 FMatrix ProjectionMatrix = FReversedZOrthoMatrix(HalfWidth, HalfHeight, 1.0f / (FarZ - NearZ), 0.0f);
450
451 // Combined view-projection matrix
452 Cascade.ViewProjectionMatrix = ViewMatrix * ProjectionMatrix;
453
454 // Update camera position/forward from component (ensure consistency)
455 Cascade.LightCameraPosition = CameraLocation;
456 Cascade.LightCameraForward = CameraRotation.Vector();
457}

462void FIVSmokeCSMRenderer::CreateCascadeRenderTargets(FIVSmokeCascadeData& Cascade, int32 Resolution)
463{
464 // Create depth render target (R32F)
465 Cascade.DepthRT = NewObject<UTextureRenderTarget2D>();
466 Cascade.DepthRT->AddToRoot(); // Prevent GC
467 Cascade.DepthRT->RenderTargetFormat = ETextureRenderTargetFormat::RTF_R32f;
468 Cascade.DepthRT->InitAutoFormat(Resolution, Resolution);
469 Cascade.DepthRT->AddressX = TextureAddress::TA_Clamp;
470 Cascade.DepthRT->AddressY = TextureAddress::TA_Clamp;
471 Cascade.DepthRT->ClearColor = FLinearColor::Black;
472 Cascade.DepthRT->UpdateResourceImmediate(true);
473
474 // Create VSM render target (RG32F) with UAV support for compute shader processing
475 const UIVSmokeSettings* Settings = UIVSmokeSettings::Get();
476 if (Settings && Settings->bEnableVSM)
477 {
478 Cascade.VSMRT = NewObject<UTextureRenderTarget2D>();
479 Cascade.VSMRT->AddToRoot();
480 Cascade.VSMRT->RenderTargetFormat = ETextureRenderTargetFormat::RTF_RG32f;
481 Cascade.VSMRT->bCanCreateUAV = true; // Required for VSM compute shader processing
482 Cascade.VSMRT->InitAutoFormat(Resolution, Resolution);
483 Cascade.VSMRT->AddressX = TextureAddress::TA_Clamp;
484 Cascade.VSMRT->AddressY = TextureAddress::TA_Clamp;
485 Cascade.VSMRT->ClearColor = FLinearColor::Black;
486 Cascade.VSMRT->UpdateResourceImmediate(true);
487 }
488
489 UE_LOG(LogIVSmokeCSM, Verbose, TEXT("[FIVSmokeCSMRenderer::CreateCascadeRenderTargets] Created RTs for cascade %d: %dx%d"),
490 Cascade.CascadeIndex, Resolution, Resolution);
491}

494{
495 if (!CaptureComponent)
496 {

500 // Orthographic projection for directional light
