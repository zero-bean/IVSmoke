14// Cascade Data Structure

16/**
17 * Data for a single shadow cascade.
18 * Contains render targets, matrices, and update state.
19 *
20 * TIMING MODEL (Single-Buffer):
21 * ============================================================================
22 * This uses a SINGLE-BUFFER approach because SceneCaptureComponent2D with
23 * bCaptureEveryFrame=true executes capture in the SAME frame's render pass.
24 *
25 * Frame N timeline:
26 * 1. Game Thread: Update() calculates VP matrix and sets CaptureComponent transform
27 * 2. Render Thread: SceneCaptureComponent2D captures depth using current transform
28 * 3. Render Thread: Ray march shader samples the captured texture with VP matrix
29 *
30 * Since capture and sampling happen in the SAME frame, no double-buffering is needed.
31 * The VP matrix calculated in Update() matches the texture captured in the same frame.
32 *
33 * NOTE: Double-buffering would be needed if capture happened at END of frame and
34 * sampling happened at START of next frame, but that's not the case here.
35 * ============================================================================
36 */
37struct IVSMOKE_API FIVSmokeCascadeData
38{
39 /** Cascade index (0 = nearest, N-1 = farthest). */
40 int32 CascadeIndex = 0;
41
42 /** View distance range for this cascade. */
43 float NearPlane = 0.0f;
44 float FarPlane = 0.0f;
45
46 //~==============================================================================
47 // Current Frame Values (used for both capture and shader sampling)
48
49 /** Orthographic projection width for this cascade. */
50 float OrthoWidth = 0.0f;
51
52 /** View-projection matrix for world-to-light-clip transform. */
53 FMatrix ViewProjectionMatrix = FMatrix::Identity;
54
55 /** Light camera position for this cascade (after texel snapping). */
56 FVector LightCameraPosition = FVector::ZeroVector;
57
58 /** Light camera forward direction (light travel direction, opposite of light source). */
59 FVector LightCameraForward = FVector(0.0f, 0.0f, -1.0f);
60
61 //~==============================================================================
62 // Resources
63
64 /** Depth render target (R32F). */
65 TObjectPtr<UTextureRenderTarget2D> DepthRT = nullptr;
66
67 /** Variance Shadow Map render target (RG32F). */
68 TObjectPtr<UTextureRenderTarget2D> VSMRT = nullptr;
69
70 /** Scene capture component for this cascade. */
71 TObjectPtr<USceneCaptureComponent2D> CaptureComponent = nullptr;
72
73 /** Whether this cascade needs capture this frame. */
74 bool bNeedsCapture = true;
75
76 /** Frame number when last captured. */
77 uint32 LastCaptureFrame = 0;
78};
