40 UFUNCTION(BlueprintPure, Category = "IVSmoke | Math")
41 static FORCEINLINE FIntVector IndexToGrid(int32 Index, const FIntVector& Resolution)
42 {
43 if (Resolution.X <= 0 || Resolution.Y <= 0 || Resolution.Z <= 0)
44 {
45 return FIntVector::ZeroValue;
46 }
47
48 const int32 BaseArea = Resolution.X * Resolution.Y;
49 const int32 Z = Index / BaseArea;
50 const int32 Remainder = Index % BaseArea;
51 const int32 Y = Remainder / Resolution.X;
52 const int32 X = Remainder % Resolution.X;
53
54 return FIntVector(X, Y, Z);
55 }

86 {

92 const int32 X = FMath::RoundToInt(LocalPos.X / VoxelSize) + CenterOffset.X;
93 const int32 Y = FMath::RoundToInt(LocalPos.Y / VoxelSize) + CenterOffset.Y;
94 const int32 Z = FMath::RoundToInt(LocalPos.Z / VoxelSize) + CenterOffset.Z;
95
96 if (X >= 0 && X < Resolution.X &&
97 Y >= 0 && Y < Resolution.Y &&
98 Z >= 0 && Z < Resolution.Z)
99 {
100 return FIntVector(X, Y, Z);
101 }
102
103 return InvalidGridPos;
104 }

143 static FORCEINLINE bool IsVoxelBitSet(const TArray<uint64>& VoxelBitArray, const FIntVector& GridPos, const FIntVector& Resolution)
144 {
145 // Resolution must be less than 64 for each axis to fit in uint64 bitfield
146 check(Resolution.X >= 0 && Resolution.X < 64 &&
147 Resolution.Y >= 0 && Resolution.Y < 64 &&
148 Resolution.Z >= 0 && Resolution.Z < 64);
149
150 const int32 Index = GridToVoxelBitIndex(GridPos, Resolution);
151
152 if (!VoxelBitArray.IsValidIndex(Index))
153 {
154 return false;
155 }
156
157 return VoxelBitArray[Index] & (1ULL << GridPos.X);
158 }
