23{

30 FVector3f SmokeColor; // 12 bytes

35 FVector3f CenterOffset; // 12 bytes
37 float DensityScale; // 4 bytes
38
40 FVector3f VolumeWorldAABBMin; // 12 bytes
41 float VoxelSize; // 4 bytes
42
44 FVector3f VolumeWorldAABBMax; // 12 bytes
45 uint32 VoxelBufferOffset; // 4 bytes
46
47 FVector3f VoxelWorldAABBMin; // 12 bytes
48 float FadeInDuration; // 4 bytes
49 FVector3f VoxelWorldAABBMax; // 12 bytes
50 float FadeOutDuration; // 4 bytes
51
52 float Reserved[4]; // 16 bytes (future use / alignment)
53};
54
